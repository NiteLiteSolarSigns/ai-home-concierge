  <script>

    function getTopCategory(subcat) {

      const map = {

        "Plumbing": "Internal",

        "Electrical": "Internal",

        "HVAC": "Internal",

        "Appliance Repair": "Internal",

        "Pest Control": "Internal",

        "Wi-Fi/Internet": "Internal",


        "Lawn Care": "External",

        "Landscaping": "External",

        "Snow Removal": "External",

        "Fence Repair": "External",

        "Sprinklers": "External",


        "Roofing": "Structure",

        "Windows & Doors": "Structure",

        "Gutters": "Structure",

        "Foundation": "Structure",

        "Security Systems": "Structure",


        "House Cleaning": "Lifestyle",

        "Babysitting": "Lifestyle",

        "Elder Care": "Lifestyle",

        "Pet Care": "Lifestyle",

        "Dining": "Lifestyle"

      };

      return map[subcat] || "";

    }


    async function loadProviders() {

      const response = await fetch('data/providers.csv');

      const text = await response.text();

      const rows = text.trim().split('\n');

      const headers = rows[0].split(',');


      const providers = rows.slice(1).map(row => {

        const values = row.split(',');

        return headers.reduce((obj, key, i) => {

          obj[key.trim()] = values[i]?.trim();

          return obj;

        }, {});

      });


      renderProviders(providers);

      populateCategoryFilter(providers);

    }


    function renderProviders(providers) {

      const container = document.getElementById('providerList');

      container.innerHTML = '';


      providers.forEach(p => {

        const topCategory = getTopCategory(p.Category);

        const card = document.createElement('div');

        card.className = `provider-card ${topCategory}`;

        card.innerHTML = `

          <h3>${p["Business Name"]}</h3>

          <p><strong>Top Category:</strong> ${topCategory}</p>

          <p><strong>Subcategory:</strong> ${p.Category}</p>

          <p><strong>Phone:</strong> ${p.Phone}</p>

          <p><strong>Email:</strong> <a href="mailto:${p.Email}">${p.Email}</a></p>

          <p><strong>Website:</strong> <a href="https://${p.Website}" target="_blank">${p.Website}</a></p>

          <p><strong>Address:</strong> ${p.Address}, ${p["Zip Code"]}</p>

          <button onclick='addToHome("${p["Business Name"]}", "${p.Category}")'>Add to Home</button>

        `;

        container.appendChild(card);

      });

    }


    function populateCategoryFilter(providers) {

      const filter = document.getElementById('categoryFilter');

      const categories = [...new Set(providers.map(p => p.Category))].sort();

      categories.forEach(cat => {

        const option = document.createElement('option');

        option.value = cat;

        option.textContent = cat;

        filter.appendChild(option);

      });


      filter.addEventListener('change', () => {

        const selected = filter.value;

        const filtered = selected

          ? providers.filter(p => p.Category === selected)

          : providers;

        renderProviders(filtered);

      });

    }


    function addToHome(name, category) {

      const selected = JSON.parse(localStorage.getItem('homeProviders') || '{}');

      selected[category] = name;

      localStorage.setItem('homeProviders', JSON.stringify(selected));

      alert(`${name} added to ${category} on your homeboard!`);

    }


    loadProviders();
